FILE(GLOB SRC *.cpp *.h *.def)

IF(WIN32)
  DECORATE_NEL_LIB("nel_drv_opengl_win")
ELSE(WIN32)
  DECORATE_NEL_LIB("nel_drv_opengl")
ENDIF(WIN32)
SET(NLDRV_OGL_LIB ${LIBNAME})
DECORATE_NEL_LIB("nel3d")
SET(NL3D_LIB ${LIBNAME})

ADD_LIBRARY(${NLDRV_OGL_LIB} SHARED ${SRC})

INCLUDE_DIRECTORIES(${LIBXML2_INCLUDE_DIR} ${OPENGL_INCLUDE_DIR})
TARGET_LINK_LIBRARIES(${NLDRV_OGL_LIB} ${NL3D_LIB} ${OPENGL_LIBRARIES})
SET_TARGET_PROPERTIES(${NLDRV_OGL_LIB} PROPERTIES VERSION ${NL_VERSION})

IF(WIN32)
  TARGET_LINK_LIBRARIES(${NLDRV_OGL_LIB} dinput.lib dxguid.lib)
  SET_TARGET_PROPERTIES(${NLDRV_OGL_LIB} PROPERTIES COMPILE_FLAGS "/Yustdopengl.h")
  SET_SOURCE_FILES_PROPERTIES(stdopengl.cpp PROPERTIES COMPILE_FLAGS "/Ycstdopengl.h")
  SET_TARGET_PROPERTIES(${NLDRV_OGL_LIB} PROPERTIES LINK_FLAGS "/NODEFAULTLIB:libcmt")
  ADD_DEFINITIONS(/DDRIVER_OPENGL_EXPORTS)
ENDIF(WIN32)

ADD_DEFINITIONS(${LIBXML2_DEFINITIONS})
