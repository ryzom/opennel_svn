FILE(GLOB SRC *.cpp *.h)

DECORATE_NEL_LIB("nel_drv_openal")
SET(NLDRV_OPENAL_LIB ${LIBNAME})
DECORATE_NEL_LIB("nelsnd_lowlevel")
SET(NLSND_LOWLEVEL_LIB ${LIBNAME})

ADD_LIBRARY(${NLDRV_OPENAL_LIB} SHARED ${SRC})

INCLUDE_DIRECTORIES(${LIBXML2_INCLUDE_DIR} ${OPENAL_INCLUDE_DIR})
TARGET_LINK_LIBRARIES(${NLDRV_OPENAL_LIB} ${LIBXML2_LIBRARIES} ${OPENAL_LIBRARY} ${NLSND_LOWLEVEL_LIB})
SET_TARGET_PROPERTIES(${NLDRV_OPENAL_LIB} PROPERTIES VERSION ${NL_VERSION})
ADD_DEFINITIONS(${LIBXML2_DEFINITIONS})

IF(WIN32)
  TARGET_LINK_LIBRARIES(${NLDRV_OPENAL_LIB} eax.lib eaxguid.lib alut.lib)
  SET_TARGET_PROPERTIES(${NLDRV_OPENAL_LIB} PROPERTIES COMPILE_FLAGS "/Yustdopenal.h")
  SET_SOURCE_FILES_PROPERTIES(stdopenal.cpp PROPERTIES COMPILE_FLAGS "/Ycstdopenal.h")
  SET_TARGET_PROPERTIES(${NLDRV_OPENAL_LIB} PROPERTIES LINK_FLAGS "/NODEFAULTLIB:libcmt")
ENDIF(WIN32)

INSTALL(TARGETS ${NLDRV_OPENAL_LIB} LIBRARY DESTINATION lib ARCHIVE DESTINATION lib)
