FILE(GLOB SRC *.cpp *.h)

DECORATE_NEL_LIB("nel3d")
SET(NL3D_LIB ${LIBNAME})
DECORATE_NEL_LIB("nelceguirenderer")
SET(NLCEGUIRENDERER_LIB ${LIBNAME})

ADD_EXECUTABLE(nl_sample_cegui ${SRC})

INCLUDE_DIRECTORIES(${LIBXML2_INCLUDE_DIR} ${CEGUI_INCLUDE_DIRS})
TARGET_LINK_LIBRARIES(nl_sample_cegui ${LIBXML2_LIBRARIES} ${CEGUI_LIBRARY} ${PLATFORM_LINKFLAGS} ${NL3D_LIB} ${NLCEGUIRENDERER_LIB})
IF(WIN32)
  SET_TARGET_PROPERTIES(nl_sample_cegui PROPERTIES LINK_FLAGS "/NODEFAULTLIB:libcmt /SUBSYSTEM:WINDOWS")
ENDIF(WIN32)
ADD_DEFINITIONS(${LIBXML2_DEFINITIONS})

INSTALL(TARGETS nl_sample_cegui RUNTIME DESTINATION bin)
INSTALL(DIRECTORY datafiles/ 
	DESTINATION share/nel/nl_sample_cegui/datafiles
	PATTERN "CVS" EXCLUDE
	PATTERN ".svn" EXCLUDE
	PATTERN "Makefile*" EXCLUDE)
